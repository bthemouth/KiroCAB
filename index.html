            function copyReport() {

                try {

                    const reportText = document.getElementById('report-text').textContent;

                    if (reportText.trim()) {

                        navigator.clipboard.writeText(reportText)

                            .then(() => alert('Relatório copiado para a área de transferência!'))

                            .catch(err => alert('Falha ao copiar relatório: ' + err));

                    } else {

                        alert('Nenhum relatório para copiar. Gere um relatório primeiro.');

                    }

                } catch (error) {

                    alert('Falha ao copiar relatório: ' + error);

                }

            }
